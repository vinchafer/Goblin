```javascript
// Navigation zwischen Seiten
document.addEventListener('DOMContentLoaded', function() {
    // Navigation-Buttons
    const navButtons = document.querySelectorAll('.nav-btn');
    const pages = document.querySelectorAll('.page');
    
    // Dark Mode Toggle
    const darkModeToggle = document.getElementById('dark-mode-toggle');
    
    // Reset Button
    const resetDataBtn = document.getElementById('reset-data-btn');
    
    // Überprüfe gespeicherten Dark-Mode
    const isDarkMode = localStorage.getItem('darkMode') === 'true';
    
    // Setze Dark Mode basierend auf localStorage
    if (isDarkMode) {
        document.body.classList.add('dark-mode');
        darkModeToggle.checked = true;
    }
    
    // Event Listener für Navigation
    navButtons.forEach(button => {
        button.addEventListener('click', function() {
            const pageId = this.getAttribute('data-page');
            
            // Aktiven Zustand aktualisieren
            navButtons.forEach(btn => btn.classList.remove('active'));
            this.classList.add('active');
            
            // Seiten anzeigen/verstecken
            pages.forEach(page => page.classList.remove('active'));
            document.getElementById(`${pageId}-page`).classList.add('active');
        });
    });
    
    // Event Listener für Dark Mode Toggle
    darkModeToggle.addEventListener('change', function() {
        if (this.checked) {
            document.body.classList.add('dark-mode');
            localStorage.setItem('darkMode', 'true');
        } else {
            document.body.classList.remove('dark-mode');
            localStorage.setItem('darkMode', 'false');
        }
    });
    
    // Event Listener für Reset Button
    resetDataBtn.addEventListener('click', function() {
        if (confirm('Möchtest du wirklich alle Daten zurücksetzen? Dies kann nicht rückgängig gemacht werden.')) {
            localStorage.removeItem('habits');
            window.location.reload();
        }
    });
    
    // Initialisiere die Gewohnheiten-Anzeige
    if (typeof initializeHabits === 'function') {
        initializeHabits();
    }
});
```

## **Architekturvorschlag: Layered Approach**

### **1. Datenebene (storage.js)**
```javascript
// Datenstruktur für eine Gewohnheit
const habitSchema = {
    id: number,
    name: string,
    icon: string,
    target: number,
    unit: string,
    frequency: 'daily' | 'weekly',
    // Streak-spezifische Felder
    streak: {
        current: number,        // Aktuelle Serie
        longest: number,        // Längste Serie
        lastCompleted: string,  // ISO-Datum des letzten Abschlusses "2024-03-15"
        history: string[]       // Array der Tage mit Abschluss ["2024-03-15", "2024-03-14"]
    }
};
```

### **2. Business Logic Layer (streakLogic.js)**
```javascript
class StreakManager {
    /**
     * Überprüft, ob ein Tag bereits gezählt wurde
     */
    static isDayAlreadyLogged(habit, date) {
        return habit.streak.history.includes(date);
    }
    
    /**
     * Überprüft, ob gestern abgeschlossen wurde (für Kontinuität)
     */
    static wasYesterdayCompleted(habit, today) {
        const yesterday = new Date(today);
        yesterday.setDate(yesterday.getDate() - 1);
        const yesterdayStr = yesterday.toISOString().split('T')[0];
        
        return habit.streak.history.includes(yesterdayStr);
    }
    
    /**
     * Aktualisiert die Streak-Logik
     */
    static updateStreak(habit, date) {
        if (this.isDayAlreadyLogged(habit, date)) {
            return habit; // Bereits gezählt, keine Änderung
        }
        
        habit.streak.history.push(date);
        habit.streak.lastCompleted = date;
        
        const yesterdayCompleted = this.wasYesterdayCompleted(habit, date);
        const twoDaysAgoCompleted = this.checkTwoDaysAgo(habit, date);
        
        if (yesterdayCompleted) {
            // Kontinuierliche Serie fortsetzen
            habit.streak.current++;
        } else if (twoDaysAgoCompleted) {
            // Ein Tag wurde verpasst, aber Serie kann gerettet werden (grace period)
            habit.streak.current = 1;
        } else {
            // Neue Serie starten
            habit.streak.current = 1;
        }
        
        // Längste Serie aktualisieren
        if (habit.streak.current > habit.streak.longest) {
            habit.streak.longest = habit.streak.current;
        }
        
        return habit;
    }
    
    /**
     * Berechnet aktuelle Serie basierend auf History
     */
    static calculateCurrentStreak(habit) {
        // Logik zur Berechnung aus der History
        const sortedDates = [...habit.streak.history].sort().reverse();
        let streak = 0;
        let currentDate = new Date();
        
        for (let i = 0; i < sortedDates.length; i++) {
            const dateStr = sortedDates[i];
            const date = new Date(dateStr);
            
            // Prüfe ob aufeinanderfolgend
            if (this.isConsecutive(currentDate, date)) {
                streak++;
                currentDate = date;
            } else {
                break;
            }
        }
        
        return streak;
    }
}
```

### **3. Service Layer (habitService.js)**
```javascript
class HabitService {
    constructor(storageKey = 'habits') {
        this.storageKey = storageKey;
    }
    
    /**
     * Markiert eine Gewohnheit für heute als abgeschlossen
     */
    completeHabit(habitId, amount = 1) {
        const habits = this.loadHabits();
        const habit = habits.find(h => h.id === habitId);
        
        if (!habit) return null;
        
        // Fortschritt aktualisieren
        habit.current = Math.min(habit.current + amount, habit.target);
        
        // Wenn Ziel erreicht, Streak aktualisieren
        const today = new Date().toISOString().split('T')[0];
        
        if (habit.current >= habit.target) {
            // Streak logik nur wenn Ziel erreicht
            const updatedHabit = StreakManager.updateStreak(habit, today);
            const index = habits.findIndex(h => h.id === habitId);
            habits[index] = updatedHabit;
        }
        
        this.saveHabits(habits);
        return habit;
    }
    
    /**
     * Gibt Streak-Statistiken zurück
     */
    getStreakStats(habitId) {
        const habits = this.loadHabits();
        const habit = habits.find(h => h.id === habitId);
        
        if (!habit) return null;
        
        return {
            current: habit.streak.current,
            longest: habit.streak.longest,
            lastCompleted: habit.streak.lastCompleted,
            historyCount: habit.streak.history.length
        };
    }
    
    /**
     * Überprüft täglichen Reset (um Mitternacht)
     */
    checkDailyReset() {
        const habits = this.loadHabits();
        const today = new Date().toISOString().split('T')[0];
        const lastReset = localStorage.getItem('lastResetDate');
        
        if (lastReset !== today) {
            // Täglicher Reset für Fortschritt (nicht für Streak!)
            habits.forEach(habit => {
                if (habit.frequency === 'daily') {
                    habit.current = 0; // Nur Fortschritt zurücksetzen
                }
            });
            
            localStorage.setItem('lastResetDate', today);
            this.saveHabits(habits);
        }
    }
}
```

```javascript
class StreakUI {
    static renderStreakBadge(habit) {
        const streak = habit.streak.current;
        let badgeClass = 'streak-badge';
        let emoji = '🔥';
        
        if (streak >= 30) {
            badgeClass += ' legendary';
            emoji = '🏆';
        } else if (streak >= 7) {
            badgeClass += ' epic';
            emoji = '⭐';
        } else if (streak >= 3) {
            badgeClass += ' good';
            emoji = '👍';
        }
        
        return `
            <div class="${badgeClass}">
                ${emoji} ${streak} Tage Serie
                <div class="streak-tooltip">
                    Aktuell: ${streak} Tage<br>
                    Rekord: ${habit.streak.longest} Tage<br>
                    Zuletzt: ${habit.streak.lastCompleted || 'Nie'}
                </div>
            </div>
        `;
    }
    
    static renderStreakCalendar(habit) {
        // 30-Tage-Kalendervisualisierung
        const history = habit.streak.history;
        const today = new Date();
        
        let calendarHTML = '<div class="streak-calendar">';
        
        for (let i = 29; i >= 0; i--) {
            const date = new Date(today);
            date.setDate(date.getDate() - i);
            const dateStr = date.toISOString().split('T')[0];
            const isCompleted = history.includes(dateStr);
            const isToday = i === 0;
            
            calendarHTML += `
                <div class="calendar-day 
                    ${isCompleted ? 'completed' : ''}
                    ${isToday ? 'today' : ''}"
                    title="${dateStr}: ${isCompleted ? 'Erledigt' : 'Nicht erledigt'}">
                </div>
            `;
        }
        
        calendarHTML += '</div>';
        return calendarHTML;
    }
}
```
